#include <stdio.h>

int main(void)
{
    int N, i, j, idx=0, count=0;
	int arr[1000];

	scanf("%d", &N);

	for(i=0; i<N; i++)
		scanf("%d", &arr[i]);

	for(i=0; i<N; i++)
	{
		if((arr[i]!=1) && (arr[i]!=2))
		{
			for(j=2; j<arr[i]; j++)
			{
				if(arr[i]%j == 0)
				{
					idx++;
					break;
				}
			}
		}

		if(arr[i] == 1)
			idx++;

		if(idx == 0)
			count++;

		idx=0;
	}

	printf("%d", count);
    
    return 0;
}
