#include <stdio.h>
#include <algorithm>

using namespace std;

int CO[1001][3];
int DP[1001][3];

int main(void) {
	int n;
	scanf("%d", &n);

	for (int i = 1; i <= n; i++)
		scanf("%d %d %d", &CO[i][0], &CO[i][1], &CO[i][2]);

	for (int i = 1; i <= n; i++) {
		DP[i][0] = min(DP[i - 1][1], DP[i - 1][2]) + CO[i][0];
		DP[i][1] = min(DP[i - 1][0], DP[i - 1][2]) + CO[i][1];
		DP[i][2] = min(DP[i - 1][0], DP[i - 1][1]) + CO[i][2];
	}

	sort(DP[n], DP[n] + 3);
	printf("%d", DP[n][0]);


	return 0;
}
