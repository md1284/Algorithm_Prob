#include <stdio.h>
#include <algorithm>
#include <windows.h>

using namespace std;

int score[10001];
int DP[10001][2];

int main(void) {
	int n;
	scanf("%d", &n);

	for (int i = 1; i <= n; i++) {
		scanf("%d", &score[i]);
	}

	for (int i = 1; i <= n; i++) {;
		DP[i][1] = max(DP[i - 2][0], DP[i - 2][1]) + score[i];
		DP[i][0] = DP[i - 1][1] + score[i];
	}

	printf("%d", DP[n][0] > DP[n][1] ? DP[n][0] : DP[n][1]);

	system("pause");
	return 0;
}
