#include <stdio.h>
#include <algorithm>
#include <windows.h>

using namespace std;

int G[51][51];
int w, h;
int cnt;

void search(void);
void inspection(int i, int j);

void search(void) {
	for (int i = 1; i <= h; i++) {
		for (int j = 1; j <= w; j++) {
			if (G[i][j] == 1) {
				inspection(i, j);
				cnt++;
			}
		}
	}
}

void inspection(int i, int j) {
	G[i][j] = 0;

	if (G[i - 1][j - 1] == 1)
		inspection(i - 1, j - 1);
	if (G[i][j - 1] == 1)
		inspection(i, j - 1);
	if (G[i + 1][j - 1] == 1)
		inspection(i + 1, j - 1);
	if (G[i - 1][j] == 1)
		inspection(i - 1, j);
	if (G[i + 1][j] == 1)
		inspection(i + 1, j);
	if (G[i - 1][j + 1] == 1)
		inspection(i - 1, j + 1);
	if (G[i][j + 1] == 1)
		inspection(i, j + 1);
	if (G[i + 1][j + 1] == 1)
		inspection(i + 1, j + 1);
}

int main(void) {
	
	while (1) {
		scanf("%d %d", &w, &h);

		if (w == 0 && h == 0)
			break;

		for (int i = 1; i <= h; i++) {
			for (int j = 1; j <= w; j++) {
				scanf("%d", &G[i][j]);
			}
		}

		search();
		printf("%d\n", cnt);

		for (int i = 1; i <= h; i++) {
			for (int j = 1; j <= w; j++) {
				G[i][j] = 0;
			}
		}
		cnt = 0;
	}

	system("pause");
	return 0;
}
