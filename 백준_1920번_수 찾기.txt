#include <stdio.h>
#include <windows.h>

#define TRUE 1
#define FALSE 0

void Margesort(int arr[], int left, int right);
void Twoarea(int arr[], int left, int mid, int right);
int Binarysearch(int arr[], int left, int right, int x);


void Twoarea(int arr[], int left, int mid, int right)
{
	int f = left;
	int r = mid + 1;
	int s = left;
	int i;
	int sortarr[500001];

	while ((f <= mid) && (r <= right))
	{
		if (arr[f] <= arr[r])
		{
			sortarr[s] = arr[f];
			f++;
		}
		else
		{
			sortarr[s] = arr[r];
			r++;
		}
		s++;
	}

	if (f > mid)
	{
		for (i = r; i <= right; i++)
		{
			sortarr[s] = arr[i];
			s++;
		}
	}
	else
	{
		for (i = f; i <= mid; i++)
		{
			sortarr[s] = arr[i];
			s++;
		}
	}

	for (i = left; i <= right; i++)
	{
		arr[i] = sortarr[i];
		sortarr[i] = 0;
	}
}

void Margesort(int arr[], int left, int right)
{
	int mid;

	if (left < right)
	{
		mid = (right + left) / 2;

		Margesort(arr, left, mid);
		Margesort(arr, mid + 1, right);

		Twoarea(arr, left, mid, right);
	}
}

int Binarysearch(int arr[], int left, int right, int x)
{
	int mid = (left + right) / 2;

	if (left <= right)
	{
		if (x < arr[mid])
			return Binarysearch(arr, left, mid - 1, x);

		else if (x > arr[mid])
			return Binarysearch(arr, mid + 1, right, x);

		else
			return TRUE;
	}

	else
		return FALSE;
}


int main(void)
{
	int arr[1000000];
	int str[1000000];
	int n1, n2, cnt, i;

	scanf("%d", &n1);

	for (i = 0; i < n1; i++)
		scanf("%d", &arr[i]);

	Margesort(arr, 0, n1 - 1);

	scanf("%d", &n2);

	for (i = 0; i < n2; i++)
	{
		scanf("%d", &cnt);

		if (Binarysearch(arr, 0, n1 - 1, cnt) == TRUE)
			str[i] = 1;
		else
			str[i] = 0;
	}

	for (i = 0; i < n2; i++)
		printf("%d\n", str[i]);

	system("pause");
	return 0;
}
