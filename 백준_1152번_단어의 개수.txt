import java.util.*;

public class Main {	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		String s = sc.nextLine();
		String[] a = s.split(" ");
		int count = a.length;
		
		for(int i=0; i<a.length; i++) {
			if(a[i].trim().length() == 0)
				count--;
		}
		System.out.println(count);
	}
}
