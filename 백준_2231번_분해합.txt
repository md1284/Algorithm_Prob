#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int sieve[1000001];

int sumation(int k) {
	int sum = 0;
	while(k > 0) {
		sum += k%10;
		k /= 10;
	}
	return sum;
}

void eratosthenes() {
	sieve[1] = 1;
	for(int i = 1; i<1000001; i++) {		
		int sum = i + sumation(i);
		
		if(sieve[sum] == 0 || sieve[sum] > i)
			sieve[sum] = i;
	}
}

int main() {	
	int n;
	cin >> n;
	
	eratosthenes();
	
	cout << sieve[n] << endl;
			
	return 0;
}
