#include <stdio.h>
#include <vector>
#include <windows.h>

using namespace std;

#define TRUE 1

typedef struct time {
	int f, r;
} Time;

Time G[100001];
int arr[100001];
int cnt;
int n, idx;
unsigned int tmin = 2147483647 + 10;

int main(void) {
	scanf("%d", &n);

	for (int i = 0;i < n;i++) {
		scanf("%d %d", &G[i].f, &G[i].r);
		if (G[i].r < G[i].f) {
			arr[i] = 1;
		}
	}

	while (TRUE) {
		for (int j = 0; j < n; j++) {
			if (arr[j] == 1)
				continue;

			if (tmin > G[j].r) {
				tmin = G[j].r;
				idx = j;
			}
			if (tmin == G[j].r) {
				if (G[j].f < G[idx].f) {
					idx = j;
				}
			}
		}
		if (tmin == 2147483647 + 10)
			break;

		arr[idx] = 1;

		for (int j = 0; j < n; j++) {
			if (arr[j] == 1)
				continue;
			if (tmin > G[j].f)
				arr[j] = 1;
		}
		cnt++;

		if (tmin == 2147483647 + 10)
			break;

		tmin = 2147483647 + 10;
	}

	system("pause");
	return 0;
}
