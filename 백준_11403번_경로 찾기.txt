#include <stdio.h>

using namespace std;

int G[101][101];
int D[101][101];
int visit[101];
int num, f;
int di, dj;

void DFS(int n) {
	if (n == dj && f != 0) {
		D[di][dj] = 1;
		return;
	}

	if (visit[n] == 1)
		return;
	else if (f != 0)
		visit[n] = 1;

	f = 1;

	for (int i = 1; i <= num; i++) {
		if (G[n][i] == 0)
			continue;
		if (D[di][dj] == 1)
			return;

		DFS(i);
	}
}


int main(void) {
	scanf("%d", &num);

	for (int i = 1; i <= num; i++) {
		for (int j = 1; j <= num; j++) {
			scanf("%d", &G[i][j]);
		}
	}
	
	for (di = 1; di <= num; di++) {
		for (dj = 1; dj <= num; dj++) {
			f = 0;

			DFS(di);

			for (int k = 1; k <= num; k++)
				visit[k] = 0;
		}
	}

	for (int i = 1; i <= num; i++) {
		for (int j = 1; j <= num; j++) {
			printf("%d ", D[i][j]);
		}
		printf("\n");
	}

	
	return 0;
}
