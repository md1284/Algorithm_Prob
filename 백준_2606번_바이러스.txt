#include <stdio.h>
#include <vector>
#include <windows.h>

using namespace std;

vector<int> G[101];
int visit[101];
int n, m;
int cnt;

void DFS(int cur) {
	visit[cur] = 1;
	cnt++;

	for (int i = 0; i < G[cur].size(); i++) {
		if (visit[G[cur][i]] == 1)
			continue;
		DFS(G[cur][i]);
	}
}

int main(void) {
	scanf("%d %d", &n, &m);

	for (int i = 0; i < m; i++) {
		int a, b;
		scanf("%d %d", &a, &b);

		G[a].push_back(b);
		G[b].push_back(a);
	}

	DFS(1);
	printf("%d", cnt-1);

	system("pause");
	return 0;
}
