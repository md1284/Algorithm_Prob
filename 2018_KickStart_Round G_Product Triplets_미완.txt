#include <iostream>
#include <fstream>
#include <vector>
#include <math.h>
#include <algorithm>

using namespace std;

typedef unsigned long long ULL;

vector<int> arr;
int c, n;

ULL solve(void) {
	ULL cnt = 0;
	
	for(int i = 0; i<n-2; i++) {
		for(int j = i+1; j<n-1; j++) {
			for(int k = j+1; k<n; k++) {
				
				if(arr[i] > 100000 && arr[j] > 100000)
					break;
				
				if((ULL)arr[k] > (ULL)arr[i]*arr[j])
					break;
				else if((ULL)arr[k] == (ULL)arr[i]*arr[j])
					cnt++;
					
			}
		}
	}
	return cnt;
}



int main() {
    ifstream inf("A-small-practice.in");
    ofstream outf("output.txt");
	
	//cin >> c;
	inf >> c;
	
    for(int i = 0; i<c; i++) {
    	//cin >> n;
    	inf >> n;
    	
    	for(int j = 0; j<n; j++) {
    		int a;
    		//cin >> a;
    		inf >> a;
			arr.push_back(a);
		}
    	
		sort(arr.begin(), arr.begin() + n);
		
		cout << i+1 << "done " << endl;
		//cout << "Case #" << i+1 << ": " << solve() << endl;
		outf << "Case #" << i+1 << ": " << solve() << endl;
		
		arr.clear();
	}
    
    inf.close();
    outf.close();
    return 0;
}
