#include <iostream>
#include <fstream>
#include <vector>
#include <math.h>

using namespace std;

int c, n, p;
vector<string> bid;

void press(string str) {
	if(bid.size() == 0) {
		bid.push_back(str);
		return;
	}

	for(int i = 0; i<bid.size(); i++) {
		int len = min(bid[i].size(), str.size());
		
		if(bid[i].substr(0, len).compare(str.substr(0, len)) == 0) {
			if(bid[i].size() >= str.size()) {
				bid.erase(bid.begin() + i);
				bid.push_back(str);
			}
			else
				return;
		}
	}
	bid.push_back(str);
	return;
}

long long int solve(void) {
	long long int sum = 0;
	
	for(int i = 0; i<bid.size(); i++)
		sum += pow(2, n-bid[i].size());
		
	return pow(2, n) - sum;
}


int main() {
    ifstream inf("A-large-practice.in");
    ofstream outf("output.txt");
	
	long long int a = pow(2, 50);
	cout << pow(2, 50) << endl;
	cout << a << endl;
	
	//scanf("%d", &c);
	inf >> c;
	
    for(int i = 0; i<c; i++) {
    	//scanf("%d %d", &n, &p);
    	inf >> n >> p;
    	
    	for(int j = 0; j<p; j++) {
    		string str;
    		//cin >> str;
    		inf >> str;
    		
    		press(str);
		}
		
		//cout << "Case #" << i+1 << ": " << solve() << endl;
		outf << "Case #" << i+1 << ": " << solve() << endl;
		bid.clear();
	}
    
    inf.close();
    outf.close();
    return 0;
}
