#include <stdio.h>
#include <vector>
#include <windows.h>

using namespace std;

int DP[1000001];

int main(void) {
	int n;
	scanf("%d", &n);
	DP[1] = 0;
	
	for (int i = 2; i <= n; i++) {
		DP[i] = DP[i - 1] + 1;
		if (i % 2 == 0 && DP[i] > DP[i / 2] + 1)
			DP[i] = DP[i / 2] + 1;
		if (i % 3 == 0 && DP[i] > DP[i / 3] + 1)
			DP[i] = DP[i / 3] + 1;
	}

	printf("%d", DP[n]);


	system("pause");
	return 0;
}
